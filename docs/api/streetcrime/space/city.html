104        self,
105        tolerance: int = 15,
106        traffic_factor: int = 1,
107        roads_file: str | None = None,
108        buildings_file: str | None = None,
109        building_categories: pd.DataFrame | None = None,
110        boundaries_file: str | None = None,
111    ) -> None:
112        """Obtains the data of the city from OSMNX and saves it in the specified files
113        or reads data from the file passed as arguments.
114
115        Parameters
116        ----------
117        tolerance : int, default = 15
118            The tolerance according to which intersectation are consolidated with `osmnx.simplification.consolidate_intersections`.
119        traffic_factor : int, default = 1
120            The factor with which speeds will be refactored. If not specified, velocities will be set to max speed per road.
121        roads_file : str, default = None
122            The path to the file containing the roads or to save the roads to if not present. If None, the default path is outputs/city/{city_name}_roads.gpkg
123        buildings_file : str, default = None
124            The path to the file containing the buildings or to save the buildings to if not present. If None, he default path is outputs/city/{city_name}_buildings.gpkg
125        building_categories : pd.DataFrame, default = None
126            The dataframe containing the categories of the buildings. Look at source code for default values.
127
128        Raises
129        ------
130        ValueError
131            If `traffic_factor` is not between 0 and 1 or if `tolerance` is not greater than 0
132        """
133
134        # Check parameters
135        if not 0 < traffic_factor <= 1:
136            raise ValueError(
137                f"Traffic factor set to {traffic_factor}. Should be between 0 and 1"
138            )
139        if tolerance <= 0:
140            raise ValueError(f"Tolerance set to {tolerance}. Should be greater than 0")
141
142        if not roads_file:
143            if not os.path.isdir("outputs/city"):
144                os.makedirs("outputs/city")
145            roads_file = f"outputs/city/{self.city_name.split(',')[0]}_roads.gpkg"
146
147        # Obtain roads
148        if os.path.isfile(roads_file):
149            print(f"Roads already downloaded in {roads_file}. Loading...")
150            if ".gpkg" in roads_file:
151                # TODO: remove useless columns
152                self.roads_nodes = gpd.read_file(roads_file, layer="nodes").set_index(
153                    "osmid"
154                )
155                self.roads_edges = gpd.read_file(roads_file, layer="edges").set_index(
156                    ["u", "v", "key"]
157                )
158                self.roads = nx.DiGraph(
159                    ox.graph_from_gdfs(self.roads_nodes, self.roads_edges)
160                )
161                self.roads_edges = (
162                    self.roads_edges.sort_values("travel_time")
163                    .reset_index()
164                    .drop(columns=["key"])
165                    .groupby(["u", "v"])
166                    .first()
167                )
168                return
169
170        roads = self._obtain_roads(tolerance, traffic_factor)
171
172        # Obtain buildings
173        buildings = self._obtain_buildings(buildings_file, building_categories)
174        buildings["nearest_node"] = buildings.geometry.apply(
175            lambda x: ox.nearest_nodes(roads, x.centroid.x, x.centroid.y)
176        )
177        buildings = buildings.groupby("nearest_node").count()
178        buildings = buildings[["home", "work", "activity", "open_day", "open_night"]]
179        nx.set_node_attributes(roads, buildings.transpose().to_dict())
180        self.roads_nodes = ox.graph_to_gdfs(roads, nodes=True, edges=False)
181        ox.save_graph_geopackage(roads, filepath=roads_file, directed=True)
182        print(f"Saved roads in {roads_file}")
183        self.roads = nx.DiGraph(roads)
184
185        # Save boundaries
186        (
187            ox.geocoder.geocode_to_gdf(self.city_name)
188            .to_crs(self.crs)
189            .to_file(f"outputs/city/{self.city_name.split(',')[0]}_boundaries.gpkg")
190        )

236    def _obtain_roads(
237        self, tolerance: int = 15, traffic_factor: int = 1
238    ) -> nx.MultiDiGraph:
239        """Download roads using `osmnx.graph_from_place`
240
241        Parameters
242        ----------
243        tolerance : int, default = 15
244            The tolerance according to which intersectation are consolidated with `osmnx.simplification.consolidate_intersections`.
245        traffic_factor : int, default = 1
246            The factor with which speeds will be refactored. If not specified, velocities will be set to max speed per road.
247
248
249        Returns
250        -------
251        `networkx.MultiDiGraph`
252            returns a networkx Multi Directed Graph with the road network of the city
253        """
254        start_time = time.time()
255        print(f"Downloading roads of {self.city_name}...")
256
257        # Downloading the full dataset of roads
258        roads = ox.graph_from_place(
259            self.city_name
260        )  # simplification is already activated
261        roads = ox.projection.project_graph(roads, to_crs=self.crs)
262
263        if tolerance > 0:
264            roads = ox.simplification.consolidate_intersections(
265                roads, tolerance=tolerance
266            )
267
268        roads = ox.speed.add_edge_speeds(roads)
269        nodes, edges = ox.graph_to_gdfs(roads, nodes=True, edges=True)
270
271        # Calculate the speed of each edge
272        if traffic_factor != 1:
273            edges["speed_kph"] = edges["speed_kph"] * traffic_factor
274            roads = ox.graph_from_gdfs(nodes, edges)
275
276        # Add travel times
277        roads = ox.speed.add_edge_travel_times(roads)
278
279        # Remove nodes without outgoing edges
280        out_degree = pd.Series(dict(roads.out_degree))
281        roads.remove_nodes_from(out_degree[out_degree == 0].index)
282
283        # Find all pairs shortest paths (faster than lazy computation at each iteration)
284        paths = dict(nx.all_pairs_dijkstra_path(roads, weight="travel_time"))
285        data = []
286        # Iterate through the nested dictionary
287        for origin, destinations in paths.items():
288            for destination, path in destinations.items():
289                # Append a tuple with the origin, destination, and path to the list
290                data.append((origin, destination, path))
291        paths = pd.DataFrame(data, columns=["node", "destination", "path"])
292        paths.astype({"node": "int64", "destination": "float64"})
293        paths.to_csv(
294            f"outputs/city/{self.city_name.split(',')[0]}_paths.csv", index=False
295        )
296        self.paths = paths
297
298        print("Downloaded roads: " + "--- %s seconds ---" % (time.time() - start_time))
299        return roads

237    def _obtain_roads(
238        self, tolerance: int = 15, traffic_factor: int = 1
239    ) -> nx.MultiDiGraph:
240        """Download roads using `osmnx.graph_from_place`
241
242        Parameters
243        ----------
244        tolerance : int, default = 15
245            The tolerance according to which intersectation are consolidated with `osmnx.simplification.consolidate_intersections`.
246        traffic_factor : int, default = 1
247            The factor with which speeds will be refactored. If not specified, velocities will be set to max speed per road.
248
249
250        Returns
251        -------
252        `networkx.MultiDiGraph`
253            returns a networkx Multi Directed Graph with the road network of the city
254        """
255        start_time = time.time()
256        print(f"Downloading roads of {self.city_name}...")
257
258        # Downloading the full dataset of roads
259        roads = ox.graph_from_place(
260            self.city_name
261        )  # simplification is already activated
262        roads = ox.projection.project_graph(roads, to_crs=self.crs)
263
264        if tolerance > 0:
265            roads = ox.simplification.consolidate_intersections(
266                roads, tolerance=tolerance
267            )
268
269        roads = ox.speed.add_edge_speeds(roads)
270        nodes, edges = ox.graph_to_gdfs(roads, nodes=True, edges=True)
271
272        # Calculate the speed of each edge
273        if traffic_factor != 1:
274            edges["speed_kph"] = edges["speed_kph"] * traffic_factor
275            roads = ox.graph_from_gdfs(nodes, edges)
276
277        # Add travel times
278        roads = ox.speed.add_edge_travel_times(roads)
279
280        # Remove nodes without outgoing edges
281        out_degree = pd.Series(dict(roads.out_degree))
282        roads.remove_nodes_from(out_degree[out_degree == 0].index)
283
284        # Find all pairs shortest paths (faster than lazy computation at each iteration)
285        paths = dict(nx.all_pairs_dijkstra_path(roads, weight="travel_time"))
286        data = []
287        # Iterate through the nested dictionary
288        for origin, destinations in paths.items():
289            for destination, path in destinations.items():
290                # Append a tuple with the origin, destination, and path to the list
291                data.append((origin, destination, path))
292        paths = pd.DataFrame(data, columns=["node", "destination", "path"])
293        paths.astype({"node": "int64", "destination": "float64"})
294        paths.to_csv(
295            f"outputs/city/{self.city_name.split(',')[0]}_paths.csv", index=False
296        )
297        self.paths = paths
298
299        print("Downloaded roads: " + "--- %s seconds ---" % (time.time() - start_time))
300        return roads
